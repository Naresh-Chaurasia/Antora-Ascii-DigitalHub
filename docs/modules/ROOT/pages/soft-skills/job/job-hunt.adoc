<script>
function toggleItem(checkbox, itemId) {
  const item = checkbox.parentElement;
  if (checkbox.checked) {
    item.classList.add('completed');
  } else {
    item.classList.remove('completed');
  }
  saveProgress();
}

function saveProgress() {
  const progress = {};
  document.querySelectorAll('.checklist-item').forEach(item => {
    const checkbox = item.querySelector('.checkbox');
    progress[item.id] = checkbox.checked;
  });
  localStorage.setItem('jobSystemProgress', JSON.stringify(progress));
}

function loadProgress() {
  const saved = localStorage.getItem('jobSystemProgress');
  if (saved) {
    const progress = JSON.parse(saved);
    Object.keys(progress).forEach(itemId => {
      const item = document.getElementById(itemId);
      if (item) {
        const checkbox = item.querySelector('.checkbox');
        checkbox.checked = progress[itemId];
        if (progress[itemId]) {
          item.classList.add('completed');
        }
      }
    });
  }
}

document.addEventListener('DOMContentLoaded', loadProgress);
</script>

<script>
function saveNotes() {
  const notes = document.getElementById('weekly-notes').value;
  if (notes.trim()) {
    const weekStart = new Date();
    weekStart.setDate(weekStart.getDate() - weekStart.getDay());
    const weekKey = `week_${weekStart.toISOString().split('T')[0]}`;
    
    const allNotes = JSON.parse(localStorage.getItem('weeklyNotes') || '{}');
    allNotes[weekKey] = {
      notes: notes,
      date: weekStart.toLocaleDateString()
    };
    
    localStorage.setItem('weeklyNotes', JSON.stringify(allNotes));
    displaySavedNotes();
    document.getElementById('weekly-notes').value = '';
  }
}

function displaySavedNotes() {
  const container = document.getElementById('saved-notes');
  const allNotes = JSON.parse(localStorage.getItem('weeklyNotes') || '{}');
  
  const sortedWeeks = Object.keys(allNotes).sort().reverse().slice(0, 4);
  
  if (sortedWeeks.length > 0) {
    container.innerHTML = '<h4>Previous Weeks:</h4>';
    sortedWeeks.forEach(weekKey => {
      const weekData = allNotes[weekKey];
      const div = document.createElement('div');
      div.className = 'journal-entry';
      div.innerHTML = `
        <div style="font-weight: 600; margin-bottom: 0.25rem;">Week of ${weekData.date}</div>
        <div style="color: #374151; line-height: 1.5;">${weekData.notes.replace(/\n/g, '<br>')}</div>
      `;
      container.appendChild(div);
    });
  }
}

document.addEventListener('DOMContentLoaded', displaySavedNotes);
</script>

== 💡 Quick Reminders

* **Quality over quantity** - Better to apply to 5 perfect jobs than 50 random ones
* **Follow up** - Send a follow-up email 1 week after applying
* **Keep learning** - Use this time to acquire new skills
* **Stay positive** - The right job is out there, keep going!

*Every application is practice. You're getting better with each one!*
